pragma solidity ^0.8.0;

contract StudentDB {
    struct Student {
        string name;
        uint256 rollno;
        string className; // Renamed to avoid conflict with the keyword 'class'
    }

    Student[] private students;

    // Function to add a student
    function addStudent(string memory name, uint256 rollno, string memory className) public {
        students.push(Student(name, rollno, className));
    }

    // Function to get student details by their ID (index in the array)
    function getStudentById(uint256 id) public view returns (string memory, uint256, string memory) {
        require(id < students.length, "Student does not exist in database");
        return (students[id].name, students[id].rollno, students[id].className);
    }

    // Function to get total number of students
    function getTotalNumberOfStudents() public view returns (uint256) {
        return students.length;
    }
}
